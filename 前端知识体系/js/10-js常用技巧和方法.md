##### 判断数组中对象的某个字段的值是否重复

```js
let arr = [
    { name:'censek', age: 18 },
    { name: 'bob', age: 13 },
    { name: 'john', age: 25 },
    { name: 'censek', age: 6 }
];
 
for (i = 0; i < arr.length - 1; i++) {
    for (j = i + 1; j < arr.length; j++) {
        if (arr[i].name === arr[j].name) {
            console.log("重名了！！");
        }
    }
}
```
